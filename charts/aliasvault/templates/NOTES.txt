AliasVault has been deployed successfully!

1. Get the application URL by running these commands:
{{- if .Values.httpRoute.enabled }}
{{- if .Values.httpRoute.hostnames }}
  export APP_HOSTNAME={{ .Values.httpRoute.hostnames | first }}
{{- else }}
  export APP_HOSTNAME=$(kubectl get --namespace {{ (first .Values.httpRoute.parentRefs).namespace | default .Release.Namespace }} gateway/{{ (first .Values.httpRoute.parentRefs).name }} -o jsonpath="{.spec.listeners[0].hostname}")
{{- end }}
  echo "Visit https://$APP_HOSTNAME to access AliasVault"
  echo "Admin panel: https://$APP_HOSTNAME/admin"
{{- else if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "aliasvault.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "aliasvault.fullname" . }} --template "{{"{{"}}range (index .status.loadBalancer.ingress 0){{"}}"}}{{"{{"}}."}}{{"}}"}}{{"}}"}}"
  echo http://$SERVICE_IP:{{ .Values.service.httpPort }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "aliasvault.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:80
  echo "Visit http://127.0.0.1:8080 to access AliasVault"
  echo "Admin panel: http://127.0.0.1:8080/admin"
{{- end }}

2. IMPORTANT - Change the default secret:
   The DATA_PROTECTION_CERT_PASSWORD is currently set to the default value.
   Update it in your values.yaml before deploying to production:

   secret:
     dataProtectionCertPassword: "your-secure-random-password"

3. Set your hostname:
   config:
     hostname: "aliasvault.yourdomain.com"

4. AliasVault runs WITHOUT built-in TLS (FORCE_HTTPS_REDIRECT=false by default).
   Use an external reverse proxy (Ingress/HTTPRoute + cert-manager) for HTTPS.

{{- if .Values.service.smtp.enabled }}
5. SMTP ports 25 and 587 are enabled. Ensure your cluster and DNS are configured
   for email delivery (public IP, MX records, PTR records).
{{- end }}
